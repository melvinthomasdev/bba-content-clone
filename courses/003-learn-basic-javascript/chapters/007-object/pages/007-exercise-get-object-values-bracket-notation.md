Read the value of the `102` property
from the `subjects` object
using the **bracket notation**.

<Editor lang="javascript" type="exercise">
<code>
let subjects = {
  101: "Physics",
  102: "Biology",
  103: "Chemistry"
};

console.log();
</code>

<solution>
let subjects = {
  101: "Physics",
  102: "Biology",
  103: "Chemistry"
};

console.log(subjects[102]);
</solution>
