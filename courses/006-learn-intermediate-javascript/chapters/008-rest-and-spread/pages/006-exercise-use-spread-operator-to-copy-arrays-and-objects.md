Copy the contents of
`evenNumbers` array to
`evenNumbersCopy` array using
the spread operator.

<Editor type="exercise" lang="javascript">
<code>
let evenNumbers = [2, 4, 6];
let evenNumbersCopy;

console.log(evenNumbersCopy);
</code>

<solution>
let evenNumbers = [2, 4, 6];
let evenNumbersCopy = [...evenNumbers];

console.log(evenNumbersCopy);
</solution>
