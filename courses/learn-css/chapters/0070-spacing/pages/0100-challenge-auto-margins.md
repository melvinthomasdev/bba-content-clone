Horizontally center the `ul` element using `auto` margin.

<codeblock language="css" type="exercise" testMode="fixedInput">
<code>
<panel language="html">
<ul>
  <li>Ice cream</li>
  <li>Chocolate</li>
  <li>Movies</li>
</ul>
</panel>
<panel language="css">
ul {
  max-width: 200px;
  background-color: tan;
}
</panel>
</code>

<solution>
ul {
  max-width: 200px;
  background-color: tan;
  margin: auto;
}
</solution>
</codeblock>
