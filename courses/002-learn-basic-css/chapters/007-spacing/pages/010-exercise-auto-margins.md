Horizontally center the `ul` element using `auto` margin.

<Editor lang="css" type="exercise">
<code>
<panel lang="html">
<ul>
  <li>Ice cream</li>
  <li>Chocolate</li>
  <li>Movies</li>
</ul>
</panel>
<panel lang="css">
ul {
  max-width: 200px;
  background-color: tan;
}
</panel>
</code>

<solution>
ul {
  max-width: 200px;
  background-color: tan;
  margin: auto;
}
</solution>
</Editor>
