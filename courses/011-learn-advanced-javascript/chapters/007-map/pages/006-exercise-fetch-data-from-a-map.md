Use the `map.get()` method
to get the value of the **name**
key.

<Editor type="exercise" lang="javascript">
<code>
let fruits = new Map([
  ["name", "apple"]
]);
let result = ;
console.log(result);
</code>

<solution>
let fruits = new Map([
  ["name", "apple"]
]);
let result = fruits.get("name");
console.log(result);
</solution>
