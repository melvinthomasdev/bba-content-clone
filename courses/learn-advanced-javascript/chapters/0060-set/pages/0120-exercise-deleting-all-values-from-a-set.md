Use the `clear()` method
to remove all the elements
from the `fruits` set.

<codeblock type="exercise" language="javascript" testMode="fixedInput">
<code>
const fruits = new Set(["apple", "banana", "orange", "mango"]);

console.log(fruits);
</code>

<solution>
const fruits = new Set(["apple", "banana", "orange", "mango"]);
fruits.clear();
console.log(fruits);
</solution>
</codeblock>
