Use HTML `<form>` element with **action** attribute set to `/signup` and **method** attribute set to `"get"`.

<Editor lang="html" type="exercise">
<code>
<form>
  <input type="email">
  <button>
    Signup
  </button>
</form>
</code>

<solution>
<form action="/signup" method="get">
  <input type="email">
  <button>
    Signup
  </button>
</form>
</solution>
</Editor>
