The `<input type="reset">` resets all the form data to their default values.

<Editor lang="html">
<code>
<form>
  <label>Username:</label>
  <input type="text">
  <br />
  <label>Password:</label>
  <input type="password">
  <br />
  <input type="submit">
  <input type="reset">
</form>
</code>
